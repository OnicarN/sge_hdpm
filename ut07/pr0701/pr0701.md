# UT07: Web Controllers y cliente Web

## controller 
```py
# -*- coding: utf-8 -*-
from odoo import http
from odoo.http import request

class TallerController(http.Controller):

 
    @http.route('/taller/bienvenida', auth='public', type='http', website=True)
    def bienvenida(self, **kw):
        return """
            <div style="text-align:center; margin-top:50px;">
                <h1>¡Bienvenido al Taller Mecánico!</h1>
                <p>Tu vehículo en las mejores manos.</p>
                <a href="/taller/vehiculos">Ver listado de vehículos</a>
            </div>
        """

    @http.route('/taller/vehiculos', auth='public', type='http', website=True)
    def listado_vehiculos(self, **kw):
        
        vehiculos = request.env['taller.vehiculo'].sudo().search([])
        
      
        return request.render('taller.listado_vehiculos_template', {
            'vehiculos': vehiculos,
        })
```
